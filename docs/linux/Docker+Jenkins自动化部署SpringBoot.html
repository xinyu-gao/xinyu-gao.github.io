# Docker + Jenkins 自动化部署 SpringBoot

# 1. 添加 Maven 插件

<plugin>
    <groupId>com.spotify</groupId>
    <artifactId>docker-maven-plugin</artifactId>
    <version>1.1.0</version>
    <executions>
        <execution>
            <id>build-image</id>
            <phase>package</phase>
            <goals>
                <goal>build</goal>
            </goals>
        </execution>
    </executions>
    <configuration>
        <imageName>${project.artifactId}:${project.version}</imageName>
        <dockerHost>http://175.24.147.229:2375</dockerHost>
        <baseImage>java:8</baseImage>
        <entryPoint>["java", "-jar", "/${project.build.finalName}.jar", "--spring.profiles.active=prod"]</entryPoint>
        <resources>
            <resource>
                <targetPath>/</targetPath>
                <directory>${project.build.directory}</directory>
                <include>${project.build.finalName}.jar</include>
            </resource>
        </resources>
    </configuration>
</plugin>

解释：

  1. imageName：Docker 镜像名。
  2. dockerHost：Docker 仓库地址，IP 地址 + 2375端口号（可以在第三步换成其他端口）。
  3. baseImage：依赖的其他镜像。
  4. entryPoint：Docker 镜像执行的命令。

# 2. Docker Registry 2.0 搭建

docker run -d -p 5000:5000 --restart=always --name registry2 registry:2

# 3. Docker 开启远程 API

参考 《Docker守护进程》 一文。

让 Docker 支持 http 上传镜像：

echo '{ "insecure-registries":["192.168.3.101:5000"] }' > /etc/docker/daemon.json

重启配置：

systemctl daemon-reload

重启 Docker：

systemctl stop docker
systemctl start docker

# 5. 配置 SSH

安装 SSH 插件：

首页 -> Manage Jenkins -> Manage Plugins -> Available -> Search：ssh -> 选择 SSH 插件，点击直接安装重启

配置 SSH：

然后 Manage Jenkins -> Configuration System -> SSH remote hosts -> 填写 Host、port（22）、用户验证（不能在此处添加就去 Manage Jenkins -> Manage Credentials 中添加）

# 6. 配置 Maven

安装 Maven 3.6.3：

# 以 Root 身份进入容器
docker exec -it --user root <container id> bash
# 下载 maven
wget https://www-us.apache.org/dist/maven/maven-3/3.6.3/binaries/apache-maven-3.6.3-bin.tar.gz
# 解压缩
tar xzvf apache-maven-3.6.3-bin.tar.gz

# 8. 编写脚本命令

在 /mydata/sh 目录下新建 detection.sh 文件，写入：

app_name="detection"
docker stop ${app_name}
echo '----stop container----'
docker rm ${app_name}
echo '----rm container----'
docker run -p 12345:12345 --name ${app_name} \
	-v /mydata/${app_name}/logs:/logs \
	-d ${app_name}:1.0.0

echo '----start container----'
