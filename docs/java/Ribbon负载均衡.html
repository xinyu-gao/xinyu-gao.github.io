import org.springframework.cloud.client.discovery.EnableDiscoveryClient;

@EnableDiscoveryClient

    public static void main(String[] args) {
        SpringApplication.run(RibbonServiceApplication.class, args);

}

# 4. 构建 restTemplate 对象

使用 @Bean 和 @LoadBalance 注解标注在 RestTemplate 上，后者注解可以使 restTemplate 对象使用负载均衡算法。

import org.springframework.cloud.client.loadbalancer.LoadBalanced;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.client.RestTemplate;

@Configuration
public class RibbonConfig {

    @Bean
    @LoadBalanced
    public RestTemplate restTemplate(){
        return new RestTemplate();
    }
}

# 6. 调用服务

@RestController
@RequestMapping("/user")
public class UserController {

    private final Logger log = LoggerFactory.getLogger(this.getClass());

    @Autowired
    private RestTemplate restTemplate;

    @Value("${service-url.user-service}")
    private String userServiceUrl;

    @GetMapping("/find")
    public String find(@RequestParam("id") Long id) {
        log.info(userServiceUrl);
        return restTemplate.getForObject(userServiceUrl + "/user/find?id={1}", String.class, id);
    }

}

# 二、启动服务

再启动一个被调用的实例，更改端口号为 8302。（即两个服务，8301、8302）如图所示：

前两个为被调用的服务，第三个为服务中心，第四个为服务消费者（即调用前两个服务）。

反复请求 http://localhost:8303/user/find?id=43112312 该地址（GET 方式），会发现 8301、8302 两个服务的方法被交替调用（交替打印 id 值）。

# 三、常见报错信息解决

常见的一个报错：No instances available for ****。

寻找三个地方是否出错：

  1. pom.xml 文件依赖是否引入有问题，spring-cloud-starter-netflix-eureka-client 为必须的依赖，其中已经包括了 Ribbon 的 jar 包文件，无需再引入其他的 Ribbon 相关依赖。
  2. 调用的 URL 地址和被调用者的应用名是否一致，是否使用了 ip:port 这种错误的方式。
  3. 查看服务消费者和被调用的服务是否都注册在注册中心。
